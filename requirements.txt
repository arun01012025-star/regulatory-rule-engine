# ── SECURITY ─────────────────────────────────────────────────────────────────
pwd_ctx = CryptContext(schemes=["bcrypt"], deprecated="auto")
bearer  = HTTPBearer()

def hash_pw(pw): return pwd_ctx.hash(pw)
def verify_pw(pw, hashed): return pwd_ctx.verify(pw, hashed)

def make_token(user_id: int, username: str, role: str) -> str:
    exp = datetime.now(timezone.utc) + timedelta(minutes=TOKEN_EXPIRE_MINUTES)
    return jwt.encode({"sub": str(user_id), "username": username, "role": role, "exp": exp}, SECRET_KEY, ALGORITHM)

def get_current_user(credentials: HTTPAuthorizationCredentials = Depends(bearer), db: Session = Depends(get_db)):
    try:
        payload = jwt.decode(credentials.credentials, SECRET_KEY, algorithms=[ALGORITHM])
        user = db.query(User).filter(User.id == int(payload["sub"])).first()
        if not user or not user.is_active:
            raise HTTPException(401, "Invalid or inactive user")
        return user
    except JWTError:
        raise HTTPException(401, "Invalid token")

def require_role(*roles):
    def checker(current_user: User = Depends(get_current_user)):
        if current_user.role not in roles:
            raise HTTPException(403, f"Access denied. Required: {list(roles)}")
        return current_user
    return checker

def log_action(db, user, action, entity_type="", entity_id="", old="", new="", ip=""):
    db.add(AuditLog(
        user_id=user.id if user else None,
        username=user.username if user else "system",
        action=action, entity_type=entity_type,
        entity_id=str(entity_id), old_value=str(old), new_value=str(new), ip_address=ip
    ))

# ── RULE ENGINE ───────────────────────────────────────────────────────────────
def run_rule_engine(data: dict) -> dict:
    quality = data.get("quality_impact", False)
    patient = data.get("patient_safety", False)
    gxpdata = data.get("gxp_record", False)
    gmp     = data.get("gmp_records", False)
    erec    = data.get("electronic_records", False)
    esig    = data.get("electronic_signatures", False)
    cloud   = data.get("cloud_hosted", False)
    rep     = data.get("critical_reports", False)
    intf    = data.get("interfaces", False)
    sys_type = data.get("system_type", "custom")
    reg_prof = data.get("regulatory_profile", "Global")
    sev     = int(data.get("severity", 1))
    prob    = int(data.get("probability", 1))
    det     = int(data.get("detectability", 1))

    gxp_relevant = quality or patient or gxpdata or gmp
    part11       = erec or esig
    annex11      = part11 and reg_prof in ("EU", "Global", "mixed")

    gamp_map = {"custom": "Category 5", "configured": "Category 4", "hybrid": "Category 4/5 Hybrid",
                "firmware": "Category 3", "infrastructure": "Category 1"}
    gamp_ref = {"custom": "GC-04", "configured": "GC-03", "hybrid": "GC-05",
                "firmware": "GC-02", "infrastructure": "GC-01"}
    gamp = gamp_map.get(sys_type, "Category 5")
    gamp_r = gamp_ref.get(sys_type, "GC-04")

    rpn = sev * prob * det
    risk_level = "Low" if rpn <= 5 else "Medium" if rpn <= 14 else "High"

    if risk_level == "High":
        test_type = "Scripted"; test_ref = "CSA-01"
    elif risk_level == "Medium":
        test_type = "Structured Unscripted"; test_ref = "CSA-03"
    else:
        test_type = "Unscripted"; test_ref = "CSA-04"

    deliverables = []
    fired = []
    if gxp_relevant:
        deliverables += ["URS", "Risk Assessment", "RTM"]
        fired.append({"id": "DV-01", "action": "GxP relevant — URS, RA, RTM required"})
    if gamp == "Category 5":
        deliverables += ["FDS", "Validation Plan", "IQ Protocol", "OQ Protocol", "PQ Protocol", "CSA Scripts"]
        fired.append({"id": "DV-02", "action": "Cat 5 — Full lifecycle deliverables"})
    if gamp == "Category 4":
        deliverables += ["Config Specification", "Testing Protocols"]
        fired.append({"id": "DV-03", "action": "Cat 4 — Config validation required"})
    if part11:
        deliverables += ["Access Control Matrix", "Audit Trail Verification", "E-Signature Verification"]
        fired.append({"id": "DV-04", "action": "Part 11 — Access/audit/e-sig deliverables"})
    if intf:
        deliverables += ["Interface Assessment", "Interface Test Scripts"]
        fired.append({"id": "DV-05", "action": "Interfaces — end-to-end testing required"})
    if rep:
        deliverables += ["Report Verification Scripts"]
        fired.append({"id": "DV-06", "action": "Critical reports — verification required"})
    if cloud:
        deliverables += ["Supplier Assessment", "Backup/DR Verification", "Security Review"]
        fired.append({"id": "DV-07", "action": "Cloud — vendor/DR/security required"})

    if quality:  fired.append({"id": "GX-01", "action": "Quality impact → GxP relevant"})
    if patient:  fired.append({"id": "GX-02", "action": "Patient safety → GxP relevant"})
    if gxpdata:  fired.append({"id": "GX-03", "action": "GxP records → validation required"})
    if erec:     fired.append({"id": "PT-01", "action": "Electronic records → record control"})
    if esig:     fired.append({"id": "PT-02", "action": "E-signatures → signature controls"})
    if cloud:    fired.append({"id": "PT-06", "action": "Cloud → vendor/DR assessment"})
    fired.append({"id": gamp_r, "action": f"{gamp} classification determined"})
    fired.append({"id": test_ref, "action": f"{test_type} testing required — RPN {rpn}"})

    approach = "Limited Documentation" if not gxp_relevant else ("CSV" if gamp == "Category 5" and risk_level == "High" else "CSA")

    return {
        "session_id": f"SES-{str(uuid.uuid4())[:5].upper()}",
        "gxp_relevant": gxp_relevant,
        "validation_required": gxp_relevant,
        "gamp_category": gamp,
        "gamp_rule_ref": gamp_r,
        "part11_applicable": part11,
        "annex11_applicable": annex11,
        "rpn": rpn,
        "risk_level": risk_level,
        "validation_approach": approach,
        "test_type": test_type,
        "test_rule_ref": test_ref,
        "required_deliverables": list(dict.fromkeys(deliverables)),
        "traceability_required": gxp_relevant,
        "fired_rules": fired,
        "template_sections": (
            (["Data Integrity Section"] if erec else []) +
            (["E-Signature Section"] if esig else []) +
            (["Design Specification Section"] if gamp == "Category 5" else []) +
            (["Challenge Testing Section"] if risk_level == "High" else []) +
            (["Interface Qualification Section"] if intf else []) +
            (["Report Verification Section"] if rep else []) +
            (["21 CFR Part 11 Compliance Section"] if part11 else []) +
            (["EU Annex 11 Compliance Section"] if annex11 else []) +
            (["Cloud/SaaS Supplier Controls Section"] if cloud else [])
        )
    }

    mp = {
        "ui_text":           {"rule": "CH-01", "impact": "No Revalidation",       "action": "UI text only — document review, no revalidation needed"},

@app.post("/api/auth/login")
def login(req: LoginReq, request: Request, db: Session = Depends(get_db)):
    user = db.query(User).filter(User.username == req.username).first()
    if not user:
        raise HTTPException(401, "Invalid credentials")
    if user.locked_until and user.locked_until > datetime.utcnow():
        raise HTTPException(423, f"Account locked until {user.locked_until.strftime('%H:%M UTC')}")
    if not verify_pw(req.password, user.hashed_password):
        user.failed_attempts += 1
        if user.failed_attempts >= 5:
            user.locked_until = datetime.utcnow() + timedelta(minutes=30)
        db.commit()
        raise HTTPException(401, "Invalid credentials")
    user.failed_attempts = 0
    user.locked_until = None
    user.last_login = datetime.utcnow()
    db.commit()
    log_action(db, user, "LOGIN", "User", user.id, ip=request.client.host)
    db.commit()
    return {"access_token": make_token(user.id, user.username, user.role),
            "token_type": "bearer", "user_id": user.id, "username": user.username,
            "full_name": user.full_name, "role": user.role, "department": user.department}
